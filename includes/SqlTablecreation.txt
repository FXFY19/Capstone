CREATE SCHEMA `capstone` ;

CREATE TABLE MTG_Card
(
  MTGCardName VARCHAR(255) NOT NULL,
  MTGCardSet VARCHAR(255) NOT NULL,
  Rarity VARCHAR(50) NOT NULL,
  Legendary VARCHAR(50),
  Cost VARCHAR(50) NOT NULL,
  Total_Cost INT NOT NULL,
  ColorID VARCHAR(255) NOT NULL,
  Card_Type1 VARCHAR(255),
  Card_Type2 VARCHAR(255),
  PRIMARY KEY (MTGCardName, MTGCardSet)
);

CREATE TABLE CFV_Card
(
  CFVCardName VARCHAR(255) NOT NULL,
  CFVCardSet VARCHAR(255) NOT NULL,
  Rarity VARCHAR(50) NOT NULL,
  Unit_Type VARCHAR(255) NOT NULL,
  Clan_Nation VARCHAR(255) NOT NULL,
  CFVTrigger VARCHAR(150),
  Grade INT NOT NULL,
  PRIMARY KEY (CFVCardName, CFVCardSet)
);

CREATE TABLE Collection
(
  Card_Name VARCHAR(255) NOT NULL,
  Card_Game_Set VARCHAR(255) NOT NULL,
  Card_Rarity CHAR(255) NOT NULL,
  Card_Game VARCHAR(255) NOT NULL,
  PRIMARY KEY (Card_Name, Card_Game_Set)
);

CREATE TABLE PKM_Card
(
  PKMName VARCHAR(255) NOT NULL,
  PKMSet VARCHAR(255) NOT NULL,
  PKMHP INT NOT NULL,
  Foil VARCHAR(255),
  FoilType VARCHAR(255),
  EnergyType VARCHAR(255),
  PKMCardType VARCHAR(255) NOT NULL,
  PKMRarity VARCHAR(255) NOT NULL,
  TrainerType VARCHAR(255),
  PRIMARY KEY (PKMName, PKMSet, PKMHP)
);

CREATE TABLE YGO_Card
(
  YGOName VARCHAR(255) NOT NULL,
  YGOSet VARCHAR(255) NOT NULL,
  YGORarity VARCHAR(255) NOT NULL,
  YGOAttribute VARCHAR(255) NOT NULL,
  YGOCardType VARCHAR(255) NOT NULL,
  YGOSub1 VARCHAR(255) NOT NULL,
  YGOPen1 VARCHAR(255),
  YGOStarRank INT,
  YGOAtk INT,
  YGODefLink INT,
  PRIMARY KEY (YGOName, YGOSet)
);
